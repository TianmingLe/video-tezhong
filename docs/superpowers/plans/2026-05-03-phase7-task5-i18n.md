- [ ] **Step 2: Implement I18nProvider**

```tsx
import React, { createContext, useMemo } from 'react'
import { dictZhCN } from './dict.zh-CN'

export type I18nValue = {
  locale: 'zh-CN'
  t: (key: string) => string
}

export const I18nContext = createContext<I18nValue | null>(null)

export function I18nProvider(props: { children: React.ReactNode }) {
  const value = useMemo<I18nValue>(() => {
    const t = (key: string) => dictZhCN[key] ?? key
    return { locale: 'zh-CN', t }
  }, [])

  return <I18nContext.Provider value={value}>{props.children}</I18nContext.Provider>
}
```

- [ ] **Step 3: Implement useT()**

```ts
import { useContext } from 'react'
import { I18nContext } from './I18nProvider'

export function useT(): (key: string) => string {
  const ctx = useContext(I18nContext)
  return ctx?.t ?? ((key: string) => key)
}
```
